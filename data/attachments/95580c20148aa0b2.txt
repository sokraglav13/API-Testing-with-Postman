      address: {

    },

        name: {type: "string"},
        catchPhhrase: {type: "string"},
        bs: {type: "string"}
    }
    },
    required: ["name", "username", "email", "address","phone","website","company","id"]
};

const responseJson = pm.response.json();

const validate = ajv.compile(schema);
